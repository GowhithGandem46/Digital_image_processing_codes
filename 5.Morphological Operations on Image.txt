import cv2 as cv

import numpy as np
img1 = cv.imread('/content/drive/MyDrive/morph1.png',0)
plt.subplot(2,4,1)
plt.axis('off')
plt.title('Original Image')
plt.imshow(img1,cmap='gray')
kernel = np.ones((5,5),np.uint8)
erosion = cv.erode(img1,kernel,iterations = 1)
plt.subplot(2,4,2)
plt.axis('off')
plt.title('Eroted Image')
plt.imshow(erosion,cmap='gray')
dilation = cv.dilate(img1,kernel,iterations = 1)
plt.subplot(2,4,3)
plt.axis('off')
plt.title('Dilated Image')
plt.imshow(dilation,cmap='gray')
opening = cv.morphologyEx(img1, cv.MORPH_OPEN, kernel)
plt.subplot(2,4,4)
plt.axis('off')
plt.title('Opening Image')
plt.imshow(opening,cmap='gray')
closing = cv.morphologyEx(img1, cv.MORPH_CLOSE, kernel)
plt.subplot(2,4,5)
plt.axis('off')
plt.title('Closing Image')
plt.imshow(closing,cmap='gray')
